class Ball {
  constructor(x, y) {
    this.x = x
    this.y = y
    this.radius = 8
    this.vx = 0
    this.vy = 0
    this.speed = 800
    this.active = false
  }
  
  launch(angle) {
    this.vx = Math.cos(angle) * this.speed
    this.vy = Math.sin(angle) * this.speed
    this.active = true
  }
  
  update(dt) {
    if (!this.active) return
    
    this.x += this.vx * dt
    this.y += this.vy * dt
    
    // è¾¹ç•Œåå¼¹
    this.checkWallCollision()
  }
  
  checkWallCollision() {
    const { width } = canvas
    
    // å·¦å³å¢™å£
    if (this.x - this.radius < 0) {
      this.x = this.radius
      this.vx = -this.vx
    }
    if (this.x + this.radius > width) {
      this.x = width - this.radius
      this.vx = -this.vx
    }
    
    // é¡¶éƒ¨
    if (this.y - this.radius < 0) {
      this.y = this.radius
      this.vy = -this.vy
    }
  }
  
  checkBrickCollision(brick) {
    // çŸ©å½¢ä¸åœ†å½¢ç¢°æ’æ£€æµ‹
    const closestX = Math.max(brick.x, Math.min(this.x, brick.x + brick.width))
    const closestY = Math.max(brick.y, Math.min(this.y, brick.y + brick.height))
    
    const distX = this.x - closestX
    const distY = this.y - closestY
    const distance = Math.sqrt(distX * distX + distY * distY)
    
    if (distance < this.radius) {
      // ç¡®å®šåå¼¹æ–¹å‘
      this.handleBrickBounce(brick, closestX, closestY)
      return true
    }
    return false
  }
  
  handleBrickBounce(brick, closestX, closestY) {
    // ç®€åŒ–çš„åå¼¹é€»è¾‘
    const fromLeft = closestX === brick.x
    const fromRight = closestX === brick.x + brick.width
    const fromTop = closestY === brick.y
    const fromBottom = closestY === brick.y + brick.height
    
    if (fromLeft || fromRight) {
      this.vx = -this.vx
    }
    if (fromTop || fromBottom) {
      this.vy = -this.vy
    }
  }
  
  render(ctx) {
    ctx.beginPath()
    ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2)
    ctx.fillStyle = '#FFFFFF'
    ctx.fill()
    ctx.closePath()
  }
}
